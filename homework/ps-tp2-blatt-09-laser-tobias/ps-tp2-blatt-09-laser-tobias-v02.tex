            \end{align*}

            \begin{solutionbox}[title=UngestÃķrte EigenzustÃĪnde]
                \begin{align*}
                    \ket{1}=\begin{pmatrix}1\\0\end{pmatrix},
                    \qquad
                    \ket{2}=\begin{pmatrix}0\\1\end{pmatrix},
                    \qquad
                    \op H_0\ket{1}=E_1^{(0)}\ket{1},
                    \qquad

            \end{solutionbox}

            \begin{calculationbox}[title=Exakte Eigenwerte]
                FÃžr die exakten Eigenwerte lÃķsen wir die charakteristische Gleichung. Dazu setzen wir

                    0

                    \det
                    \begin{pmatrix}
                        E_1^{(0)}+\lambda-E & \lambda\\
                        \lambda & E_2^{(0)}+\lambda-E

                    (E_1^{(0)}+\lambda-E)(E_2^{(0)}+\lambda-E)-\lambda^2.
                \end{align*}
                Um die quadratische Gleichung Ãžbersichtlich zu schreiben, verwenden wir
                \begin{align*}

                    \begin{align*}

                        \bar E^{(0)}+\lambda

                        \sqrt{\left(\frac{\Delta}{2}\right)^2+\lambda^2}.
                    \end{align*}
                \end{solutionbox}
            \end{calculationbox}

            \begin{calculationbox}[title=Exakte EigenzustÃĪnde]
                Wir schreiben einen Eigenzustand als Linearkombination der BasiszustÃĪnde:
                \begin{align*}

                \end{align*}

                \end{align*}

                    \end{align*}
                \end{solutionbox}
            \end{calculationbox}
        \end{calculationbox}

                \end{solutionbox}
            \end{calculationbox}
        \end{calculationbox}

        \item Nehmen Sie an, dass $E_1^{(0)}$ und $E_2^{(0)}$ fast entartet sind, d.h. $\abs{E_1^{(0)}-E_2^{(0)}}\ll \lambda$, und berechnen Sie die Energieeigenwerte bis zu erster Ordnung in $\lambda$, sowie die EnergieeigenzustÃĪnde nullter Ordnung mit StÃķrungstheorie im Fall $E_1^{(0)}=E_2^{(0)}$.             \end{calculationbox}
        \end{calculationbox}

    \paragraph{22. Magnetische Dipolwechselwirkung:}

    Wir betrachten zwei Spin-$\frac12$-Teilchen mit magnetischem Moment
    \begin{align*}
        \vec M^{(i)}=\gamma\vec S^{(i)},
        \qquad i=1,2,
    \end{align*}
    in einem externen Magnetfeld
    \begin{align*}
        \vec B=B_0\vec e_z.
    \end{align*}
    Der Hamiltonoperator des Systems ist
    \begin{align*}
        \op H_0=-\vec B\cdot(\vec M^{(1)}+\vec M^{(2)}).
    \end{align*}

    \begin{enumerate}[label=(\alph*)]
        \item Zeigen Sie, dass der Hamiltonoperator $\op H_0$ diagonal in der Produktbasis
        \begin{align*}
            \ket{++},\quad \ket{+-},\quad \ket{-+},\quad \ket{--}
        \end{align*}
        ist und geben Sie die zugehÃķrigen Energieeigenwerte an. Gibt es entartete Energieniveaus? Es ist nÃžtzlich, die Frequenz $\omega_0=\gamma B_0$ zu definieren.

        \begin{calculationbox}
            Da das Magnetfeld in $z$-Richtung zeigt, enthÃĪlt $\op H_0$ nur $z$-Komponenten der Spins. Mit $\omega_0=\gamma B_0$ gilt
            \begin{align*}
                \op H_0
                =
                -B_0\gamma(\op S_z^{(1)}+\op S_z^{(2)})
                =
                -\omega_0(\op S_z^{(1)}+\op S_z^{(2)}).
            \end{align*}
            Die Produktbasis besteht aus EigenzustÃĪnden von $\op S_z^{(1)}$ und $\op S_z^{(2)}$. Deshalb ist $\op H_0$ in dieser Basis diagonal.

            \begin{calculationbox}[title=Energieeigenwerte]
                FÃžr Spin-$\frac12$ gilt
                \begin{align*}
                    \op S_z\ket+=\frac{\hbar}{2}\ket+,
                    \qquad
                    \op S_z\ket-=-\frac{\hbar}{2}\ket-.
                \end{align*}
                Daher
                \begin{align*}
                    \op H_0\ket{++}
                    &=-\omega_0\left(\frac{\hbar}{2}+\frac{\hbar}{2}\right)\ket{++}

                    \op H_0\ket{+-}

                    \op H_0\ket{-+}
                    &=-\omega_0\left(-\frac{\hbar}{2}+\frac{\hbar}{2}\right)\ket{-+}
                    =0,\\
                    \op H_0\ket{--}

                \end{align*}

                \begin{solutionbox}[title=UngestÃķrtes Spektrum]

                        \qquad
                        E_{+-}^{(0)}=0,
                        \qquad
                        E_{-+}^{(0)}=0,
                        \qquad

                    \end{align*}
                \end{solutionbox}
                Das Niveau $E=0$ ist zweifach entartet, weil die ZustÃĪnde $\ket{+-}$ und $\ket{-+}$ dieselbe Energie besitzen.
            \end{calculationbox}
        \end{calculationbox}

        \item Die Teilchen sind sich so nahe, dass die Dipol-Dipol-Wechselwirkung der beiden magnetischen Momente nicht vernachlÃĪssigt werden kann. Diese wird durch einen Hamiltonoperator
        \begin{align*}

        \begin{align*}

        \end{align*}
        den Einheitsvektor in diese Richtung bezeichnet. Der Abstand $r$ sei groÃ genug, sodass $\op H_1$ als StÃķrungsterm behandelt werden kann.         \begin{align*}

        \end{align*}
        zu definieren.

        \begin{calculationbox}
            Wir setzen $\vec M^{(i)}=\gamma\vec S^{(i)}$ ein. Dann wird
            \begin{align*}
                \op H_1
                =
                \frac{\mu_0\gamma^2}{4\pi r^3}
                \left[
                    \vec S^{(1)}\cdot\vec S^{(2)}

            \end{align*}

                    \begin{align*}

                        \frac{\hbar\Omega}{4}(1-3\cos^2\theta).
                    \end{align*}
                \end{solutionbox}
            \end{calculationbox}
        \end{calculationbox}

        \item Berechnen Sie nun die Verschiebung der Energieniveaus $\ket{+-}$, $\ket{-+}$ in erster Ordnung von $\op H_1$ und bestimmen Sie die zugehÃķrigen Spinwellenfunktionen.
